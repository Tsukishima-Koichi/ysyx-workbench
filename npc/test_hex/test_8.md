# --- 1. 初始化基址与立即数 ---
80000000: 800012b7    lui t0, 0x80001      # t0 = 0x80001000 (测试用的内存基址)
80000004: 00a00313    li  t1, 10           # t1 = 10
80000008: 01400393    li  t2, 20           # t2 = 20

# --- 2. 算术运算与写入内存 ---
8000000c: 00730e33    add t3, t1, t2       # t3 = 10 + 20 = 30
80000010: 01c2a023    sw  t3, 0(t0)        # mem[0x80001000] = 30 (0x0000001E)
80000014: 00628223    sb  t1, 4(t0)        # mem[0x80001004] = 10 (0x0A)

# --- 3. 从内存读回数据 ---
80000018: 0002ae83    lw  t4, 0(t0)        # t4 = 30
8000001c: 0042cf03    lbu t5, 4(t0)        # t5 = 10

# --- 4. 关键验证与 a0 寄存器计算 ---
80000020: 01ee8533    add a0, t4, t5       # a0 = 30 + 10 = 40
80000024: fd800f93    li  t6, -40          # t6 = -40 (0xFFFFFFD8)
80000028: 01f50533    add a0, a0, t6       # a0 = 40 + (-40) = 0  🌟(决定生死的a0)

# --- 5. 跳转与结束停机 ---
8000002c: 80000337    lui t1, 0x80000      # t1 = 0x80000000
80000030: 03c30313    addi t1, t1, 60      # t1 = 0x8000003C
80000034: 00030067    jalr zero, 0(t1)     # 无条件跳转到 0x8000003C
80000038: 00000013    nop                  # (这条会被 Flush)
8000003c: 00100073    ebreak               # 触发停机，C++ 检查 a0



RISC-V 寄存器别名与 ABI 对应关系表
========================================================================
寄存器编号 | ABI 别名 | 描述 (Description)            | 保存者 (Saver)
------------------------------------------------------------------------
x0         | zero     | 硬连线零 (Hard-wired zero)     | ----
x1         | ra       | 返回地址 (Return address)      | 调用者 (Caller)
x2         | sp       | 栈指针 (Stack pointer)         | 被调用者 (Callee)
x3         | gp       | 全局指针 (Global pointer)      | ----
x4         | tp       | 线程指针 (Thread pointer)      | ----
x5         | t0       | 临时寄存器 0                   | 调用者 (Caller)
x6         | t1       | 临时寄存器 1                   | 调用者 (Caller)
x7         | t2       | 临时寄存器 2                   | 调用者 (Caller)
x8         | s0/fp    | 保存寄存器 0 / 帧指针           | 被调用者 (Callee)
x9         | s1       | 保存寄存器 1                   | 被调用者 (Callee)
x10        | a0       | 函数参数 0 / 返回值 0           | 调用者 (Caller)
x11        | a1       | 函数参数 1 / 返回值 1           | 调用者 (Caller)
x12        | a2       | 函数参数 2                     | 调用者 (Caller)
x13        | a3       | 函数参数 3                     | 调用者 (Caller)
x14        | a4       | 函数参数 4                     | 调用者 (Caller)
x15        | a5       | 函数参数 5                     | 调用者 (Caller)
x16        | a6       | 函数参数 6                     | 调用者 (Caller)
x17        | a7       | 函数参数 7                     | 调用者 (Caller)
x18        | s2       | 保存寄存器 2                   | 被调用者 (Callee)
x19        | s3       | 保存寄存器 3                   | 被调用者 (Callee)
x20        | s4       | 保存寄存器 4                   | 被调用者 (Callee)
x21        | s5       | 保存寄存器 5                   | 被调用者 (Callee)
x22        | s6       | 保存寄存器 6                   | 被调用者 (Callee)
x23        | s7       | 保存寄存器 7                   | 被调用者 (Callee)
x24        | s8       | 保存寄存器 8                   | 被调用者 (Callee)
x25        | s9       | 保存寄存器 9                   | 被调用者 (Callee)
x26        | s10      | 保存寄存器 10                  | 被调用者 (Callee)
x27        | s11      | 保存寄存器 11                  | 被调用者 (Callee)
x28        | t3       | 临时寄存器 3                   | 调用者 (Caller)
x29        | t4       | 临时寄存器 4                   | 调用者 (Caller)
x30        | t5       | 临时寄存器 5                   | 调用者 (Caller)
x31        | t6       | 临时寄存器 6                   | 调用者 (Caller)
------------------------------------------------------------------------
注：
1. Caller (调用者保存)：函数调用后不保证值不变，如需使用需在调用前压栈。
2. Callee (被调用者保存)：子函数在返回前必须恢复这些寄存器的原始值。

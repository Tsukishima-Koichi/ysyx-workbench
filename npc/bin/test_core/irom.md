# ==========================================
# 综合冒险与分支预测深度测试用例 (51条指令)
# ==========================================

# --- 1. U型指令测试 ---
0x00: lui   x1, 0x12345       # x1 = 0x12345000
0x04: auipc x2, 0             # x2 = PC + 0 = 0x80000004

# --- 2. 算术逻辑与连续数据冒险前递(Forwarding)测试 ---
0x08: addi  x3, x0, 15        # x3 = 15
0x0C: add   x4, x3, x3        # EX-EX 前递: x4 = 30
0x10: sub   x5, x4, x3        # EX-EX 前递: x5 = 15
0x14: slli  x6, x5, 2         # x6 = 60
0x18: slti  x7, x6, 100       # x7 = 1
0x1C: sltiu x8, x6, 100       # x8 = 1
0x20: xori  x9, x3, 15        # x9 = 0
0x24: ori   x10, x3, 240      # x10 = 255 (0xFF)
0x28: andi  x11, x10, 15      # x11 = 15
0x2C: srai  x12, x10, 4       # x12 = 15 (算术右移)
0x30: srli  x13, x10, 4       # x13 = 15 (逻辑右移)

# --- 3. 访存与 Load-Use 冒险阻塞测试 ---
0x34: lui   x14, 0x80100      # x14 = 0x80100000 (DRAM 基址)
0x38: sw    x6, 0(x14)        # M[0] = 60
0x3C: sh    x3, 4(x14)        # M[4] = 15
0x40: sb    x7, 6(x14)        # M[6] = 1
0x44: lw    x15, 0(x14)       # x15 = 60
0x48: add   x16, x15, x15     # ⚠️ LOAD-USE 冒险! 强制 stall 气泡, 算完 x16 = 120
0x4C: lh    x17, 4(x14)       # x17 = 15
0x50: lhu   x18, 4(x14)       # x18 = 15
0x54: lb    x19, 6(x14)       # x19 = 1
0x58: lbu   x20, 6(x14)       # x20 = 1

# --- 4. 分支预测器 BHT/BTB 训练与冲刷测试 ---
0x5C: addi  x21, x0, 0        # 循环变量 i = 0
0x60: addi  x22, x0, 4        # 循环上限 = 4
loop: (0x64)
0x64: addi  x21, x21, 1       # i++
0x68: blt   x21, x22, loop    # 向后跳转(-4). 前2次会预测失败，后1次预测成功，退出时再次失败冲刷

# --- 5. 剩余控制流与无条件跳转测试 ---
0x6C: beq   x21, x22, +8      # x21==4, 跳转到 0x74
0x70: addi  x23, x0, 999      # (被跳过)
0x74: bne   x21, x0, +8       # 跳转到 0x7C
0x78: addi  x23, x0, 999      # (被跳过)
0x7C: bge   x21, x22, +8      # 跳转到 0x84
0x80: addi  x23, x0, 999      # (被跳过)
0x84: bltu  x0, x21, +8       # 0 < 4, 跳转到 0x8C
0x88: addi  x23, x0, 999      # (被跳过)
0x8C: bgeu  x21, x0, +8       # 跳转到 0x94
0x90: addi  x23, x0, 999      # (被跳过)

# JAL 与 JALR
0x94: jal   x24, +12          # 跳转到 0xA0. x24 保存返回地址 0x80000098
0x98: addi  x25, x0, 999      # (被跳过)
0x9C: addi  x25, x0, 999      # (被跳过)
0xA0: jalr  x26, x24, 12      # 跳到 x24+12 = 0x800000A4. x26保存 0x800000A4
0xA4: nop

# --- 6. 剩余 R 型指令全集 ---
0xA8: sll   x27, x3, x7       # 15 << 1 = 30
0xAC: slt   x28, x3, x4       # 15 < 30 -> 1
0xB0: sltu  x29, x3, x4       # 15 < 30 -> 1
0xB4: xor   x30, x3, x4       # 15 ^ 30 = 17
0xB8: srl   x31, x4, x7       # 30 >> 1 = 15
0xBC: sra   x23, x4, x7       # 30 >> 1 = 15 (算术右移)
0xC0: or    x25, x3, x4       # 15 | 30 = 31
0xC4: and   x26, x3, x4       # 15 & 30 = 14

# --- 7. 终点 ---
0xC8: jal   x0, 0             # 停机死循环
